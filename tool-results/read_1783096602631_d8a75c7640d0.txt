   825→export function LanguageProvider({ children }: { children: React.ReactNode }) {
   826→  const [lang, setLangState] = React.useState<Lang>('en')
   827→  const [mounted, setMounted] = React.useState(false)
   828→
   829→  React.useEffect(() => {
   830→    try {
   831→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
   832→      if (saved === 'en' || saved === 'bn') setLangState(saved)
   833→    } catch {}
   834→    setMounted(true)
   835→  }, [])
   836→
   837→  const setLang = React.useCallback((l: Lang) => {
   838→    setLangState(l)
   839→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
   840→    document.documentElement.lang = l
   841→  }, [])
   842→
   843→  const toggle = React.useCallback(() => {
   844→    setLang(lang === 'en' ? 'bn' : 'en')
   845→  }, [lang, setLang])
   846→
   847→  const t = React.useCallback((key: string) => {
   848→    return translations[lang][key] ?? translations.en[key] ?? key
   849→  }, [lang])
   850→
   851→  const trFn = React.useCallback((en: string) => {
   852→    return tr(en, lang)
   853→  }, [lang])
   854→
   855→  const value = React.useMemo(() => ({ lang, setLang, toggle, t, tr: trFn }), [lang, setLang, toggle, t, trFn])
   856→
   857→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
   858→}
   859→
   860→export function useLang() {
   861→  const ctx = React.useContext(LanguageContext)
   862→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
   863→  return ctx
   864→}
