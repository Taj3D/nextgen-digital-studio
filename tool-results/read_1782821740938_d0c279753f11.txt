  1123→export function LanguageProvider({ children }: { children: React.ReactNode }) {
  1124→  const [lang, setLangState] = React.useState<Lang>('en')
  1125→  const [mounted, setMounted] = React.useState(false)
  1126→
  1127→  React.useEffect(() => {
  1128→    try {
  1129→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
  1130→      if (saved === 'en' || saved === 'bn') setLangState(saved)
  1131→    } catch {}
  1132→    setMounted(true)
  1133→  }, [])
  1134→
  1135→  const setLang = React.useCallback((l: Lang) => {
  1136→    setLangState(l)
  1137→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
  1138→    document.documentElement.lang = l
  1139→  }, [])
  1140→
  1141→  const toggle = React.useCallback(() => {
  1142→    setLang(lang === 'en' ? 'bn' : 'en')
  1143→  }, [lang, setLang])
  1144→
  1145→  const t = React.useCallback((key: string) => {
  1146→    return translations[lang][key] ?? translations.en[key] ?? key
  1147→  }, [lang])
  1148→
  1149→  const value = React.useMemo(() => ({ lang, setLang, toggle, t }), [lang, setLang, toggle, t])
  1150→
  1151→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
  1152→}
  1153→
  1154→export function useLang() {
  1155→  const ctx = React.useContext(LanguageContext)
  1156→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
  1157→  return ctx
  1158→}
