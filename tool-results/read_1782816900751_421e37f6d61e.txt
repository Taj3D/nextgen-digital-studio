   997→export function LanguageProvider({ children }: { children: React.ReactNode }) {
   998→  const [lang, setLangState] = React.useState<Lang>('en')
   999→  const [mounted, setMounted] = React.useState(false)
  1000→
  1001→  React.useEffect(() => {
  1002→    try {
  1003→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
  1004→      if (saved === 'en' || saved === 'bn') setLangState(saved)
  1005→    } catch {}
  1006→    setMounted(true)
  1007→  }, [])
  1008→
  1009→  const setLang = React.useCallback((l: Lang) => {
  1010→    setLangState(l)
  1011→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
  1012→    document.documentElement.lang = l
  1013→  }, [])
  1014→
  1015→  const toggle = React.useCallback(() => {
  1016→    setLang(lang === 'en' ? 'bn' : 'en')
  1017→  }, [lang, setLang])
  1018→
  1019→  const t = React.useCallback((key: string) => {
  1020→    return translations[lang][key] ?? translations.en[key] ?? key
  1021→  }, [lang])
  1022→
  1023→  const value = React.useMemo(() => ({ lang, setLang, toggle, t }), [lang, setLang, toggle, t])
  1024→
  1025→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
  1026→}
  1027→
  1028→export function useLang() {
  1029→  const ctx = React.useContext(LanguageContext)
  1030→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
  1031→  return ctx
  1032→}
