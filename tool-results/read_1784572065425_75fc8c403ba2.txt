  1307→export function LanguageProvider({ children }: { children: React.ReactNode }) {
  1308→  const [lang, setLangState] = React.useState<Lang>('en')
  1309→  const [mounted, setMounted] = React.useState(false)
  1310→
  1311→  React.useEffect(() => {
  1312→    let initialLang: Lang = 'en'
  1313→    try {
  1314→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
  1315→      if (saved === 'en' || saved === 'bn') {
  1316→        setLangState(saved)
  1317→        initialLang = saved
  1318→      }
  1319→    } catch {}
  1320→    // Keep the <html lang> attribute in sync with the active language on first mount
  1321→    // (so SSR 'en' is corrected to 'bn' if the user previously toggled to Bengali).
  1322→    if (typeof document !== 'undefined') {
  1323→      document.documentElement.lang = initialLang
  1324→    }
  1325→    setMounted(true)
  1326→  }, [])
  1327→
  1328→  // Keep <html lang> attribute in sync whenever lang changes (toggle / setLang calls).
  1329→  React.useEffect(() => {
  1330→    if (typeof document !== 'undefined') {
  1331→      document.documentElement.lang = lang
  1332→    }
  1333→  }, [lang])
  1334→
  1335→  const setLang = React.useCallback((l: Lang) => {
  1336→    setLangState(l)
  1337→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
  1338→    document.documentElement.lang = l
  1339→  }, [])
  1340→
  1341→  const toggle = React.useCallback(() => {
  1342→    setLang(lang === 'en' ? 'bn' : 'en')
  1343→  }, [lang, setLang])
  1344→
  1345→  const t = React.useCallback((key: string) => {
  1346→    return translations[lang][key] ?? translations.en[key] ?? key
  1347→  }, [lang])
  1348→
  1349→  const trFn = React.useCallback((en: string) => {
  1350→    return tr(en, lang)
  1351→  }, [lang])
  1352→
  1353→  const value = React.useMemo(() => ({ lang, setLang, toggle, t, tr: trFn }), [lang, setLang, toggle, t, trFn])
  1354→
  1355→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
  1356→}
  1357→
  1358→export function useLang() {
  1359→  const ctx = React.useContext(LanguageContext)
  1360→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
  1361→  return ctx
  1362→}
