   525→export function LanguageProvider({ children }: { children: React.ReactNode }) {
   526→  const [lang, setLangState] = React.useState<Lang>('en')
   527→  const [mounted, setMounted] = React.useState(false)
   528→
   529→  React.useEffect(() => {
   530→    try {
   531→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
   532→      if (saved === 'en' || saved === 'bn') setLangState(saved)
   533→    } catch {}
   534→    setMounted(true)
   535→  }, [])
   536→
   537→  const setLang = React.useCallback((l: Lang) => {
   538→    setLangState(l)
   539→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
   540→    document.documentElement.lang = l
   541→  }, [])
   542→
   543→  const toggle = React.useCallback(() => {
   544→    setLang(lang === 'en' ? 'bn' : 'en')
   545→  }, [lang, setLang])
   546→
   547→  const t = React.useCallback((key: string) => {
   548→    return translations[lang][key] ?? translations.en[key] ?? key
   549→  }, [lang])
   550→
   551→  const trFn = React.useCallback((en: string) => {
   552→    return tr(en, lang)
   553→  }, [lang])
   554→
   555→  const value = React.useMemo(() => ({ lang, setLang, toggle, t, tr: trFn }), [lang, setLang, toggle, t, trFn])
   556→
   557→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
   558→}
   559→
   560→export function useLang() {
   561→  const ctx = React.useContext(LanguageContext)
   562→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
   563→  return ctx
   564→}
