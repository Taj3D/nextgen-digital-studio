   279→export function LanguageProvider({ children }: { children: React.ReactNode }) {
   280→  const [lang, setLangState] = React.useState<Lang>('en')
   281→  const [mounted, setMounted] = React.useState(false)
   282→
   283→  React.useEffect(() => {
   284→    try {
   285→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
   286→      if (saved === 'en' || saved === 'bn') setLangState(saved)
   287→    } catch {}
   288→    setMounted(true)
   289→  }, [])
   290→
   291→  const setLang = React.useCallback((l: Lang) => {
   292→    setLangState(l)
   293→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
   294→    document.documentElement.lang = l
   295→  }, [])
   296→
   297→  const toggle = React.useCallback(() => {
   298→    setLang(lang === 'en' ? 'bn' : 'en')
   299→  }, [lang, setLang])
   300→
   301→  const t = React.useCallback((key: string) => {
   302→    return translations[lang][key] ?? translations.en[key] ?? key
   303→  }, [lang])
   304→
   305→  const trFn = React.useCallback((en: string) => {
   306→    return tr(en, lang)
   307→  }, [lang])
   308→
   309→  const value = React.useMemo(() => ({ lang, setLang, toggle, t, tr: trFn }), [lang, setLang, toggle, t, trFn])
   310→
   311→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
   312→}
   313→
   314→export function useLang() {
   315→  const ctx = React.useContext(LanguageContext)
   316→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
   317→  return ctx
   318→}
