   740→function tr(en: string, lang: Lang): string {
   741→  if (lang === 'en') return en
   742→  return contentBn[en] ?? en
   743→}

   745→export function LanguageProvider({ children }: { children: React.ReactNode }) {
   746→  const [lang, setLangState] = React.useState<Lang>('en')
   747→  const [mounted, setMounted] = React.useState(false)
   748→
   749→  React.useEffect(() => {
   750→    try {
   751→      const saved = localStorage.getItem(STORAGE_KEY) as Lang | null
   752→      if (saved === 'en' || saved === 'bn') setLangState(saved)
   753→    } catch {}
   754→    setMounted(true)
   755→  }, [])
   756→
   757→  const setLang = React.useCallback((l: Lang) => {
   758→    setLangState(l)
   759→    try { localStorage.setItem(STORAGE_KEY, l) } catch {}
   760→    document.documentElement.lang = l
   761→  }, [])
   762→
   763→  const toggle = React.useCallback(() => {
   764→    setLang(lang === 'en' ? 'bn' : 'en')
   765→  }, [lang, setLang])
   766→
   767→  const t = React.useCallback((key: string) => {
   768→    return translations[lang][key] ?? translations.en[key] ?? key
   769→  }, [lang])
   770→
   771→  const trFn = React.useCallback((en: string) => {
   772→    return tr(en, lang)
   773→  }, [lang])
   774→
   775→  const value = React.useMemo(() => ({ lang, setLang, toggle, t, tr: trFn }), [lang, setLang, toggle, t, trFn])
   776→
   777→  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>
   778→}
   779→
   780→export function useLang() {
   781→  const ctx = React.useContext(LanguageContext)
   782→  if (!ctx) throw new Error('useLang must be used within LanguageProvider')
   783→  return ctx
   784→}
